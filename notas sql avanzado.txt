SELECT c.nombre, p.fecha_pedido, pr.nombre, pr.precio_unidad  
FROM pedidos p
LEFT JOIN clientes c
ON c.id_cliente = p.cliente_id
LEFT JOIN productos pr
ON p.producto_id = pr.id_producto;
